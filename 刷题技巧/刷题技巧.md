  ```java
      ListNode next;
  }
  
  // 迭代访问
  void traverse(ListNode node){
      for(ListNode head = node;

      }
  }
  
  // 递归访问
  void traverse(ListNode node){
      
      if (node == null){
          return node;
      }
      else{
          traverse(node.next);
      }
  }
  ```

  

+ **递归翻转链表**

  + **代码**

    ```java
    class ListNode{
        int val;
        ListNode next;
        ListNode(int x){this.val = x;}
    }
    
    
    ListNode reverse(ListNode head) {
        // 递归终止条件
        if (head.next == null) return head;
        
        // 翻转 head.next 为起点的链表 
        // ( 链表递归固定套路 ) 
        ListNode last = reverse(head.next);
        
        // 递归之后的处理
        head.next.next = head;
        head.next = null;
        return last;
    }
    ```

  + **解释**

    > 1. 递归终止条件
    >
    >    > 翻转 `只有一个结点的链表` , 等于翻转自己
    >
    > 2. 翻转 `head.next 为起点的链表`
